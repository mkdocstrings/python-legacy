117def sort_object(obj: CollectorItem, sort_function: Callable[[CollectorItem], Any]) -> None: 

118 """Sort the collected object's children. 

119 

120 Sorts the object's children list, then each category separately, and then recurses into each. 

121 

122 Arguments: 

123 obj: The collected object, as a dict. Note that this argument is mutated. 

124 sort_function: The sort key function used to determine the order of elements. 

125 """ 

126 obj["children"].sort(key=sort_function) 

127 

128 for category in ("attributes", "classes", "functions", "methods", "modules"): 

129 obj[category].sort(key=sort_function) 

130 

131 for child in obj["children"]: 

132 sort_object(child, sort_function=sort_function)

135def _sort_key_alphabetical(item: CollectorItem) -> Any: 

136 # chr(sys.maxunicode) is a string that contains the final unicode 

137 # character, so if 'name' isn't found on the object, the item will go to 

138 # the end of the list. 

139 return item.get("name", chr(sys.maxunicode))

142def _sort_key_source(item: CollectorItem) -> Any: 

143 # if 'line_start' isn't found on the object, the item will go to 

144 # the start of the list. 

145 return item.get("source", {}).get("line_start", -1)
